# VPS Configuration Commands
# Run these commands manually on VPS via SSH

# 1. Check environment variables
cd /opt/stillme && env | grep -E '(ROUTING|PLACEHOLDER|MODEL|PROVIDER|API_KEY)' | head -20

# 2. Check config files
cd /opt/stillme && find . -name '*.py' -o -name '*.env' -o -name '*.yaml' -o -name '*.yml' | grep -E '(gateway|config|env)' | head -10

# 3. Check gateway service status
systemctl status stillme-gateway --no-pager

# 4. Check AI server status
systemctl status stillme-ai-server --no-pager

# 5. Check gateway logs
journalctl -u stillme-gateway -n 20 --no-pager

# 6. Check AI server logs
journalctl -u stillme-ai-server -n 20 --no-pager

# 7. Check if .env file exists
ls -la /opt/stillme/.env

# 8. Check gateway config
cat /opt/stillme/real_stillme_gateway.py | grep -E '(ROUTING|PLACEHOLDER|MODEL|PROVIDER)' | head -10
